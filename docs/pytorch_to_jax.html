Mental Model Shift

Three conceptual differences matter before any code comparison.

PyTorch
Eager execution · Mutable tensors · autograd on tensors · Global seed via torch.manual_seed · Python for loop unrolls steps
JAX / Flax NNX
Traced/compiled · Immutable arrays · Functional transforms (jax.grad) · Explicit PRNG keys · lax.scan compiles the loop
PyTorch — gradient flow
loss.backward() accumulates into .grad; must call zero_grad() before each step
JAX — gradient flow
nnx.value_and_grad(fn)(model, batch) returns a fresh gradient pytree; no accumulation, no zero_grad
Block 1 Module & Parameter Definition

Both frameworks use a class-based module. The differences: no super().__init__() in NNX, parameters use nnx.Param(array) instead of nn.Parameter(tensor), and random initialisation requires an explicit rngs handle instead of a global seed.

PyTorch
import torch
import torch.nn as nn

class VanillaRNNCell(nn.Module):
    def __init__(
        self, input_size, hidden_size, alpha=0.2
    ):
        super().__init__()        # required
        self.hidden_size = hidden_size
        self.alpha = alpha
        std = 1.0 / (hidden_size ** 0.5)

        self.W    = nn.Parameter(
            torch.randn(hidden_size, hidden_size) * std
        )
        self.W_in = nn.Parameter(
            torch.randn(hidden_size, input_size) * std
        )
        self.b    = nn.Parameter(
            torch.zeros(hidden_size)
        )

    def forward(self, h, x):
        pre = (
            torch.tanh(h) @ self.W.T
            + x @ self.W_in.T
            + self.b
        )
        return (1.0 - self.alpha) * h + self.alpha * pre
JAX / Flax NNX
import jax.numpy as jnp
import flax.nnx as nnx

class VanillaRNNCell(nnx.Module):
    def __init__(
        self, input_size, hidden_size,
        rngs: nnx.Rngs,             # explicit PRNG handle
        alpha=0.2
    ):
        # no super().__init__() needed
        self.hidden_size = hidden_size
        self.alpha = alpha
        init = nnx.initializers.normal(
            stddev=1.0 / (hidden_size ** 0.5)
        )
        self.W    = nnx.Param(
            init(rngs.params(), (hidden_size, hidden_size))
        )
        self.W_in = nnx.Param(
            init(rngs.params(), (hidden_size, input_size))
        )
        self.b    = nnx.Param(jnp.zeros((hidden_size,)))

    def __call__(self, h, x):       # forward → __call__
        pre = (
            jnp.tanh(h) @ self.W.T
            + x @ self.W_in.T
            + self.b
        )
        return (1.0 - self.alpha) * h + self.alpha * pre
Block 2 Sequential Computation — Python Loop vs lax.scan

This is the most consequential difference. PyTorch dispatches T GPU kernels (one per step). JAX's lax.scan compiles the entire loop into a single XLA operation — no per-step launch overhead. vmap then parallelises over the batch dimension.

Why nnx.split / nnx.merge?
lax.scan requires a pure function with no mutable state. nnx.split extracts the parameter pytree as an immutable snapshot captured in the closure. nnx.merge reconstructs a temporary module inside the scan body without re-allocating memory.
PyTorch — Python for-loop
class VanillaRNN(nn.Module):
    def __init__(self, input_size, hidden_size, alpha=0.2):
        super().__init__()
        self.cell = VanillaRNNCell(input_size, hidden_size, alpha)
        self.hidden_size = hidden_size

    def forward(self, xs, h0=None):
        B, T, _ = xs.shape
        h = (
            torch.zeros(B, self.hidden_size, device=xs.device)
            if h0 is None else h0
        )
        outputs = []
        for t in range(T):          # T kernel dispatches
            h = self.cell(h, xs[:, t])
            outputs.append(h)
        # stack list → tensor
        return torch.stack(outputs, dim=1), h
JAX — vmap + lax.scan
class VanillaRNN(nnx.Module):
    def __init__(self, input_size, hidden_size, rngs, alpha=0.2):
        self.cell = VanillaRNNCell(input_size, hidden_size, rngs, alpha)
        self.hidden_size = hidden_size

    def __call__(self, xs, h0=None):
        B, T, _ = xs.shape
        if h0 is None:
            h0 = jnp.zeros((B, self.hidden_size))

        # extract params once; constant across vmap + scan
        graphdef, state = nnx.split(self.cell)

        def run_single(x_seq, h_init):
            # processes ONE sequence: x_seq (T, I), h_init (H,)
            def step(h, x):
                cell = nnx.merge(graphdef, state)
                return cell(h, x), cell(h, x)   # carry, output
            h_final, outputs = jax.lax.scan(
                step, h_init, x_seq             # 1 kernel dispatch
            )
            return outputs, h_final

        # vmap maps run_single over the batch axis
        return jax.vmap(run_single)(xs, h0)     # (B,T,H), (B,H)
Block 3 Model Instantiation & PRNG

PyTorch uses a global seed. JAX threads explicit PRNG keys through every call, making randomness reproducible and composable with jit / vmap.

    input_size=1,
    hidden_size=64,
    output_size=1,

import jax

model = VanillaRNNModel(

Block 4 Training Step

PyTorch requires the zero-grad / backward / step ritual. In JAX, nnx.value_and_grad computes loss and gradients in one call; optimizer.update applies the update. No gradient accumulation means no zero_grad.

PyTorch
import torch.optim as optim

optimizer = optim.Adam(model.parameters(), lr=1e-3)

def train_step(model, optimizer, xs, targets):
    optimizer.zero_grad()           # 1. clear old grads
    preds = model(xs)
    loss = ((preds - targets) ** 2).mean()
    loss.backward()                 # 2. accumulate grads
    optimizer.step()                # 3. apply update
    return loss.item()
JAX / Flax NNX
import optax

optimizer = nnx.Optimizer(
    model, optax.adam(1e-3), wrt=nnx.Param
)

def loss_fn(model, batch):
    preds = model(batch["inputs"])
    return jnp.mean((preds - batch["targets"]) ** 2)

@nnx.jit
def train_step(model, optimizer, batch):
    loss, grads = nnx.value_and_grad(loss_fn)(model, batch)
    optimizer.update(model, grads)  # mutates model in-place
    return loss                     # no zero_grad needed
Block 5 JIT Compilation

Both frameworks support JIT. JAX traces through Python on the first call for a given input shape and caches the compiled binary; avoid data-dependent Python if / for inside JIT — use lax.cond / lax.scan instead.

PyTorch
# torch.compile wraps the whole model (PyTorch 2.0+)
model = torch.compile(model)

# Training loop is unchanged
for xs, targets in dataloader:
    loss = train_step(model, optimizer, xs, targets)
JAX / Flax NNX
# @nnx.jit decorates the FUNCTION, not the model.
# Compiled on first call; reused for same shapes.
@nnx.jit
def train_step(model, optimizer, batch):
    loss, grads = nnx.value_and_grad(loss_fn)(model, batch)
    optimizer.update(model, grads)
    return loss

# First call triggers compilation (slow).
# All subsequent calls use the cached binary (fast).
for batch in batches:
    loss = train_step(model, optimizer, batch)
Block 6 Optimizer with Gradient Clipping

PyTorch clips gradients as a separate call between backward and step. Optax composes transforms with optax.chain — clipping is baked into the optimizer definition, so train_step is unchanged.

PyTorch
optimizer = optim.Adam(model.parameters(), lr=1e-3)

def train_step(model, optimizer, xs, targets):
    optimizer.zero_grad()
    loss = ((model(xs) - targets) ** 2).mean()
    loss.backward()
    # clip before step
    torch.nn.utils.clip_grad_norm_(
        model.parameters(), max_norm=1.0
    )
    optimizer.step()
    return loss.item()
JAX / Optax
tx = optax.chain(
    optax.clip_by_global_norm(1.0),   # clip first
    optax.adam(1e-3),                  # then update
)
optimizer = nnx.Optimizer(model, tx, wrt=nnx.Param)

# train_step is IDENTICAL to Block 4 — no change needed
@nnx.jit
def train_step(model, optimizer, batch):
    loss, grads = nnx.value_and_grad(loss_fn)(model, batch)
    optimizer.update(model, grads)   # clipping applied here
    return loss
Full Training Loop

Everything assembled end-to-end.

PyTorch
import torch, torch.nn as nn, torch.optim as optim

torch.manual_seed(42)
model = VanillaRNNModel(
    input_size=1, hidden_size=64, output_size=1
)
optimizer = optim.Adam(model.parameters(), lr=1e-3)

for step in range(500):
    xs, targets = get_batch()      # (B, T, 1)
    optimizer.zero_grad()
    loss = ((model(xs) - targets) ** 2).mean()
    loss.backward()
    optimizer.step()
    if (step + 1) % 50 == 0:
        print(f"step {step+1}  loss={loss.item():.4f}")
JAX / Flax NNX
import jax, jax.numpy as jnp, flax.nnx as nnx, optax

model = VanillaRNNModel(
    input_size=1, hidden_size=64, output_size=1,
    rngs=nnx.Rngs(params=jax.random.PRNGKey(42)),
)
optimizer = nnx.Optimizer(model, optax.adam(1e-3), wrt=nnx.Param)

def loss_fn(model, batch):
    return jnp.mean((model(batch["inputs"]) - batch["targets"]) ** 2)

@nnx.jit
def train_step(model, optimizer, batch):
    loss, grads = nnx.value_and_grad(loss_fn)(model, batch)
    optimizer.update(model, grads)
    return loss

key = jax.random.PRNGKey(0)
for step in range(500):
    key, subkey = jax.random.split(key)
    batch = get_batch(subkey)      # pass key explicitly
    loss = train_step(model, optimizer, batch)
    if (step + 1) % 50 == 0:
        print(f"step {step+1}  loss={float(loss):.4f}")
